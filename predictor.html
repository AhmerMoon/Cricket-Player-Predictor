from sklearn.tree import DecisionTreeRegressor
import difflib

def find_best_match(user_input, valid_options):
    """Find best match ignoring case, spaces etc. Returns None if no good match found."""
    user_input_clean = user_input.lower().replace(" ", "")
    # Clean valid options similarly
    cleaned_options = {opt: opt.lower().replace(" ", "") for opt in valid_options}
    # Find close matches based on cleaned names
    matches = difflib.get_close_matches(user_input_clean, cleaned_options.values(), n=1, cutoff=0.6)
    if matches:
        # Return original column name corresponding to the cleaned match
        for orig, clean in cleaned_options.items():
            if clean == matches[0]:
                return orig
    return None
